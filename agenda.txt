import os
from supabase import create_client
from dotenv import load_dotenv

load_dotenv()

# ✅ Correto — carrega do .env
url = os.getenv("SUPABASE_URL")
key = os.getenv("SUPABASE_KEY")

if not url or not key:
    raise RuntimeError("SUPABASE_URL e SUPABASE_KEY são obrigatórios no .env")

supabase = create_client(url, key)


def salvar_no_supabase(nome, servico, horario):
    dados = {"cliente": nome, "servico": servico, "horario": horario}
    supabase.table("agendamentos").insert(dados).execute()


def ler_do_supabase():
    resposta = supabase.table("agendamentos").select("*").execute()
    return resposta.data
